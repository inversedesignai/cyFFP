\section{Introduction}
%=======================================================================

We compute the point spread function (PSF) of rotationally symmetric optical
systems at oblique incidence angle~$\alpha$.  The method:
\begin{enumerate}[nosep]
  \item Decomposes the near field into angular-momentum eigenmodes $e^{im\theta}$.
  \item Computes scalar spectral coefficients via Hankel transform at order~$m$.
  \item Propagates each coefficient to the focal plane via the known $e^{ik_z z}$ phase.
  \item Re-expands in a local cylindrical basis centered at the focal spot using
        Graf's addition theorem.
  \item Reconstructs the PSF by inverse Hankel transform and angular IFFT.
\end{enumerate}
The formulation is exact (no paraxial approximation).

\paragraph{Scalar formulation.}
For a linearly polarized field $\E = u(r,\theta)\,\hat p$ (where $\hat p$ is a
fixed Cartesian unit vector such as $\yhat$ or $\xhat$), the Cartesian component
along $\hat p$ satisfies the \emph{scalar} wave equation exactly.  The spectral
coefficient for each angular mode is a \emph{single} Hankel transform at order~$m$
(see \S\ref{sec:scalar_coeffs}).  This is both simpler and more correct than the
vector TE/TM approach, which uses 4~transforms at orders $m\pm 1$ per mode and
conflates different Bessel structures when the TE and TM coefficients are naively
combined.  The TE/TM decomposition is retained in Appendix~\ref{app:TETM} for
reference and for future vector-field extensions.

\section{Preliminaries}
\label{sec:prelim}
%=======================================================================

\subsection{Coordinate conventions}

Cylindrical coordinates $(r,\theta,z)$ relate to Cartesian via
$x = r\cos\theta$, $y = r\sin\theta$.
The unit vectors satisfy
\begin{equation}
  \rhat = \cos\theta\,\xhat + \sin\theta\,\yhat,\qquad
  \that = -\sin\theta\,\xhat + \cos\theta\,\yhat,
\end{equation}
and conversely $\xhat = \cos\theta\,\rhat - \sin\theta\,\that$,\;
$\yhat = \sin\theta\,\rhat + \cos\theta\,\that$.

\subsection{Near-field modal decomposition}

The tangential electric field at $z=0$ is written
\begin{equation}\label{eq:nearfield}
  \E^{\mathrm{Near}}(r,\theta)
  = \sum_m \E^{\mathrm{Near}}_m(r)\,e^{im\theta},
  \qquad
  \E^{\mathrm{Near}}_m(r) = E_{m,r}(r)\,\rhat + E_{m,\theta}(r)\,\that.
\end{equation}

\paragraph{Primary workflow: modes from FDTD/RCWA.}
For a rotationally symmetric structure, FDTD and RCWA can solve Maxwell's equations
independently for each angular-momentum index $m$.  The output of each simulation is
directly the modal near field $\E^{\mathrm{Near}}_m(r)$ --- even when the incident
field (oblique plane wave) is not rotationally symmetric.  In this workflow, the
modes $E_{m,r}(r)$, $E_{m,\theta}(r)$ are \textbf{available for each $m$ separately}
and there is no need to construct or Fourier-transform a 2D field $\E(r,\theta)$.
The appropriate entry point is \texttt{compute\_scalar\_coeffs}, which accepts
the scalar modes $u_m(r)$ for $m \ge 0$ directly (see~\eqref{eq:uy_from_cyl}
for how to extract $u_m$ from the cylindrical FDTD output).

\paragraph{Secondary workflow: analytical or synthetic 2D fields.}
When a closed-form expression for $\E(r,\theta)$ is available (e.g.\ for validation),
the modal amplitudes are extracted by discrete Fourier transform over $\theta$:
\begin{equation}
  E_{m,r/\theta}(r_j)
  = \frac{1}{N_\theta}\sum_{l=0}^{N_\theta-1}
    E_{r/\theta}(r_j,\theta_l)\,e^{-im\theta_l}.
\end{equation}
This is Step~1 (angular FFT) and is needed only in this secondary workflow.

\paragraph{Analytical decomposition for specific field forms.}
Some near fields admit closed-form $E_m(r)$ without requiring an FFT over $\theta$:

\begin{itemize}[nosep]
  \item \textbf{LPA / normal lens $+$ oblique wave:}
    $\E = t(r)\,e^{ik_x r\cos\theta}\,\hat p$, where $\hat p$ is the
    polarization vector ($\yhat$ for $s$-pol, $\xhat$ for $p$-pol).
    The Jacobi--Anger expansion gives an analytical $E_m(r)$
    (see \S\ref{sec:neumann}).  This is the fast-path case.

  \item \textbf{Ideal oblique lens:}
    $u(r,\theta) = \exp(-ik[\sqrt{(r\cos\theta-x_0)^2 + r^2\sin^2\theta + f^2} - f])$.
    The $\theta$-dependence is inside a square root, so \textbf{no closed-form
    Jacobi--Anger decomposition exists}.  Computing $E_m(r)$ requires either
    (a)~numerical FFT over $\theta$ at each $r_j$, or
    (b)~numerical quadrature of $\frac{1}{2\pi}\int_0^{2\pi} u(r,\theta)\,e^{-im\theta}\,d\theta$.
    This form is used for validation only; in practice, a real metalens designed
    for oblique incidence would be simulated in FDTD (primary workflow), which
    directly provides $E_m(r)$.
\end{itemize}

\subsection{Bessel recurrence relations}

We use the NIST identities (DLMF \S10.6) throughout:
\begin{align}
  \frac{2m}{x}\,J_m(x) &= J_{m-1}(x) + J_{m+1}(x), \label{eq:rec1}\\[4pt]
  2\,\frac{dJ_m(x)}{dx} &= J_{m-1}(x) - J_{m+1}(x), \label{eq:rec2}\\[4pt]
  J_{-m}(x) &= (-1)^m\,J_m(x). \label{eq:negm}
\end{align}
Immediate consequences with $x = k_r r$:
\begin{align}
  \frac{m}{k_r r}\,J_m(k_r r) &= \tfrac{1}{2}\bigl[J_{m-1}(k_r r) + J_{m+1}(k_r r)\bigr],
  \label{eq:moverx}\\[4pt]
  \frac{\partial J_m(k_r r)}{\partial r}
  &= \frac{k_r}{2}\bigl[J_{m-1}(k_r r) - J_{m+1}(k_r r)\bigr].
  \label{eq:dJdr}
\end{align}

\subsection{Bessel orthogonality}

\begin{equation}\label{eq:bessel_orth}
  \int_0^\infty r\,J_m(k_r r)\,J_m(k_r' r)\,dr
  = \frac{\delta(k_r - k_r')}{k_r}.
\end{equation}

\section{Scalar Wave Equation and Spectral Coefficients}
\label{sec:scalar_coeffs}
%=======================================================================

\subsection{Each Cartesian component satisfies the scalar Helmholtz equation}

\begin{tcolorbox}[title=Theorem]
In a source-free homogeneous medium, every Cartesian component of $\E$ independently
satisfies the scalar Helmholtz equation $\nabla^2 u + k^2 u = 0$.
\end{tcolorbox}

\begin{proof}
In a source-free homogeneous region ($\rho_{\mathrm{free}}=0$, constant $\epsilon,\mu$),
Maxwell's curl equations give
\begin{equation}\label{eq:maxwell_curl}
  \nabla\times\nabla\times\E = k^2\E, \qquad k = \omega\sqrt{\mu\epsilon}.
\end{equation}
The vector identity $\nabla\times\nabla\times\E = \nabla(\nabla\cdot\E) - \nabla^2\E$
combined with the divergence condition $\nabla\cdot\E = 0$ (from $\rho_{\mathrm{free}}=0$
and homogeneity) yields the \textbf{vector Helmholtz equation}:
\begin{equation}\label{eq:vector_helmholtz}
  \nabla^2\E + k^2\E = \mathbf{0}.
\end{equation}
The vector Laplacian of $\E = E_x\,\xhat + E_y\,\yhat + E_z\,\zhat$ in Cartesian
coordinates is simply
\begin{equation}
  \nabla^2\E = (\nabla^2 E_x)\,\xhat + (\nabla^2 E_y)\,\yhat + (\nabla^2 E_z)\,\zhat,
\end{equation}
because $\xhat$, $\yhat$, $\zhat$ are \emph{constant} --- they do not depend on
position.  Substituting into~\eqref{eq:vector_helmholtz} and projecting onto each
basis vector:
\begin{equation}\label{eq:scalar_helmholtz_components}
  \nabla^2 E_x + k^2 E_x = 0, \qquad
  \nabla^2 E_y + k^2 E_y = 0, \qquad
  \nabla^2 E_z + k^2 E_z = 0.
\end{equation}
Each Cartesian component independently satisfies the scalar Helmholtz equation.
\end{proof}

\paragraph{Why this does NOT hold for cylindrical components.}
The same argument fails for $E_r$ and $E_\theta$ because the cylindrical unit vectors
$\rhat(\theta)$ and $\that(\theta)$ depend on position.  Writing
$\E = E_r\,\rhat + E_\theta\,\that + E_z\,\zhat$ and applying $\nabla^2$
produces cross-derivative terms:
\begin{equation}
  \nabla^2\E \;\ne\; (\nabla^2 E_r)\,\rhat + (\nabla^2 E_\theta)\,\that + (\nabla^2 E_z)\,\zhat.
\end{equation}
Explicitly, the $r$- and $\theta$-components of $\nabla^2\E$ are (Jackson \S3.6):
\begin{align}
  (\nabla^2\E)_r &= \nabla^2 E_r - \frac{E_r}{r^2} - \frac{2}{r^2}\frac{\partial E_\theta}{\partial\theta},
  \label{eq:vec_lap_r}\\[4pt]
  (\nabla^2\E)_\theta &= \nabla^2 E_\theta - \frac{E_\theta}{r^2} + \frac{2}{r^2}\frac{\partial E_r}{\partial\theta}.
  \label{eq:vec_lap_theta}
\end{align}
The extra terms couple $E_r$ and $E_\theta$, so neither satisfies a scalar equation
alone.  This is precisely why the TE/TM decomposition (Appendix~\ref{app:TETM})
introduces the vector cylindrical harmonics with their complicated $m\pm 1$ Bessel
structure --- it diagonalizes the coupled system~\eqref{eq:vec_lap_r}--\eqref{eq:vec_lap_theta}.

\subsection{Application to linearly polarized near fields}
\label{sec:scalar_application}

Let $u(r,\theta,z) \equiv E_y(r,\theta,z)$ denote the $y$-component of the electric
field.  By~\eqref{eq:scalar_helmholtz_components}, $u$ satisfies the scalar Helmholtz
equation
\begin{equation}\label{eq:scalar_helmholtz}
  \nabla^2 u + k^2 u = 0
\end{equation}
everywhere in the propagation region $z > 0$.

\paragraph{What this means for CyFFP.}
Given the near-field $u(r,\theta,0) = E_y(r,\theta,0)$ at $z=0$, we can propagate $u$
to the focal plane $z = f$ using scalar cylindrical harmonics (Bessel functions $\times$
angular modes $\times$ plane-wave phases).  The result is $E_y(\rho,\psi,f)$, the
$y$-component of the PSF.  The PSF intensity for $y$-polarized light is $|E_y|^2$.

\paragraph{Important subtlety: the full vector field is NOT purely $\yhat$-polarized.}
Even if the near field is purely $y$-polarized at $z = 0$, the propagated field at
$z = f$ generally has nonzero $E_x$ and $E_z$ components (required by
$\nabla\cdot\E = 0$).  However, each of these components \emph{also} satisfies its
own scalar Helmholtz equation and can be propagated independently.  The scalar CyFFP
pipeline computes $E_y$ at the focal plane --- the dominant polarization component.
For the cross-polarization component $E_x$, see the discussion in
Appendix~\ref{app:TETM}.

\paragraph{Practical note: extracting $u$ from cylindrical FDTD output.}
FDTD in cylindrical coordinates outputs $E_{m,r}(r)$ and $E_{m,\theta}(r)$ for each
angular mode~$m$.  The $y$-component is obtained via
$\yhat = \sin\theta\,\rhat + \cos\theta\,\that$:
\begin{equation}\label{eq:uy_from_cyl}
  u_m(r) = E_{m,y}(r)
  = \frac{1}{2i}\bigl[E_{m-1,r}(r) - E_{m+1,r}(r)\bigr]
  + \frac{1}{2}\bigl[E_{m-1,\theta}(r) + E_{m+1,\theta}(r)\bigr],
\end{equation}
using $\sin\theta = (e^{i\theta}-e^{-i\theta})/(2i)$ and
$\cos\theta = (e^{i\theta}+e^{-i\theta})/2$, which shift the mode index by $\pm 1$.
For validation with analytical 2D fields, $u(r,\theta)$ is simply the known
$y$-component evaluated on the grid.

\subsection{Cylindrical expansion of the scalar field}

The scalar Helmholtz equation~\eqref{eq:scalar_helmholtz} in cylindrical coordinates
admits solutions $J_m(k_r r)\,e^{im\theta}\,e^{ik_z z}$ with $k_r^2 + k_z^2 = k^2$.
The near field is expanded as
\begin{equation}\label{eq:scalar_expansion}
  u(r,\theta,0) = \sum_m u_m(r)\,e^{im\theta}
  = \sum_m\int_0^\infty a_m(k_r)\,J_m(k_r r)\,k_r\,dk_r\;\,e^{im\theta},
\end{equation}
where $a_m(k_r)$ is the scalar spectral coefficient and the $k_r\,dk_r$ measure
comes from the Bessel orthogonality relation~\eqref{eq:bessel_orth}.

\subsection{Extracting the spectral coefficient}

Multiply both sides of $u_m(r) = \int_0^\infty a_m(k_r)\,J_m(k_r r)\,k_r\,dk_r$
by $J_m(k_r' r)\,r$ and integrate over $r$:
\begin{align}
  \int_0^\infty u_m(r)\,J_m(k_r' r)\,r\,dr
  &= \int_0^\infty a_m(k_r)\Bigl[\int_0^\infty J_m(k_r r)\,J_m(k_r' r)\,r\,dr\Bigr]\,k_r\,dk_r
  \nonumber\\
  &= \int_0^\infty a_m(k_r)\,\frac{\delta(k_r - k_r')}{k_r}\,k_r\,dk_r
  = a_m(k_r').
\end{align}

\begin{tcolorbox}[title=Scalar spectral coefficient]
\begin{equation}\label{eq:am_scalar}
  a_m(k_r)
  = \int_0^\infty u_m(r)\,J_m(k_r r)\,r\,dr
  = \HT_m[r\,u_m](k_r),
\end{equation}
\end{tcolorbox}

\noindent
where $\HT_m[f](k_r) = \int_0^\infty f(r)\,J_m(k_r r)\,dr$ denotes the order-$m$
Hankel transform (plain $dr$ measure, with the $r$ weight absorbed into the argument).

\paragraph{Mode count.}
Each mode~$m$ requires a \textbf{single Hankel transform} at order~$m$.
Compare with the TE/TM approach (Appendix~\ref{app:TETM}), which requires
\textbf{4~transforms} at orders $m\pm 1$ per mode.

\paragraph{Mode truncation.}
$|m| \le M_{\max} = \lceil k\sin\alpha\cdot R\rceil + \delta M$,
determined by $J_m(k\sin\alpha\cdot R) \approx 0$ for $m \gg k\sin\alpha\cdot R$.

\section{Negative-$m$ Symmetry (Scalar Case)}
\label{sec:scalar_symmetry}
%=======================================================================

We prove that $u(r,-\theta) = u(r,\theta)$ for the $y$-component of the electric
field of a rotationally symmetric structure illuminated by a $y$-polarized plane
wave propagating in the $xz$-plane.

Consider the mirror reflection $\sigma_y: (x,y,z) \to (x,-y,z)$.  A polar vector
$\E = (E_x,E_y,E_z)$ transforms as
\begin{equation}
  \sigma_y[\E](x,y,z)
  = \bigl(E_x(x,-y,z),\;-E_y(x,-y,z),\;E_z(x,-y,z)\bigr).
\end{equation}
The incident field $\E_{\mathrm{inc}} = E_0\,e^{ik_x x}\,\yhat$ satisfies
$\sigma_y[\E_{\mathrm{inc}}] = -\E_{\mathrm{inc}}$ (the spatial phase is independent
of $y$, while the $\yhat$-component picks up a minus sign from the reflection).
For a structure with $\epsilon(x,-y,z) = \epsilon(x,y,z)$, Maxwell's equations
are invariant under $\sigma_y$, so $\sigma_y[\E]$ solves the same equation with
source $\sigma_y[\E_{\mathrm{inc}}] = -\E_{\mathrm{inc}}$.  By linearity and
uniqueness:
\begin{equation}
  \sigma_y[\E] = -\E.
\end{equation}
Taking the $y$-component: $-E_y(x,-y,z) = -E_y(x,y,z)$, hence
\begin{equation}
  E_y(x,-y,z) = E_y(x,y,z).
\end{equation}
Since $u \equiv E_y$ and $\theta \to -\theta$ corresponds to $y \to -y$ at fixed $r$,
we have $u(r,-\theta) = u(r,\theta)$.

Taking the Fourier coefficient:
\begin{equation}
  u_{-m}(r) = \frac{1}{2\pi}\int_0^{2\pi} u(r,\theta)\,e^{im\theta}\,d\theta
  = \frac{1}{2\pi}\int_0^{2\pi} u(r,-\theta')\,e^{-im\theta'}\,d\theta'
  = u_m(r),
\end{equation}
where we substituted $\theta' = -\theta$.

\subsection{From modal symmetry to spectral symmetry}

Since the spectral coefficient uses the Hankel transform at order~$m$
(see~\eqref{eq:am_scalar}), the Bessel reflection $J_{-m}(x) = (-1)^m J_m(x)$
introduces a sign:
\begin{align}
  a_{-m}(k_r)
  &= \int_0^\infty u_{-m}(r)\,J_{-m}(k_r r)\,r\,dr
  = \int_0^\infty u_m(r)\,(-1)^m\,J_m(k_r r)\,r\,dr
  = (-1)^m\,a_m(k_r).
\end{align}
Since propagation is a pointwise multiplication by $e^{ik_z f}$ (same for all $m$):

\begin{tcolorbox}[title=Scalar symmetry]
\begin{equation}\label{eq:scalar_sym}
  u_{-m} = u_m
  \qquad\Longrightarrow\qquad
  a_{-m}(k_r) = (-1)^m\,a_m(k_r)
  \qquad\Longrightarrow\qquad
  \tilde a_{-m}(k_r) = (-1)^m\,\tilde a_m(k_r).
\end{equation}
\end{tcolorbox}

\noindent
The $(-1)^m$ factor is \emph{not} a peculiarity of the TE/TM decomposition ---
it is inherent in the cylindrical harmonic basis via
$J_{-m} = (-1)^m J_m$.  The scalar formulation still gives simpler symmetry
than the TE/TM case, where the TE and TM coefficients have \emph{different}
$(-1)^m$ vs $(-1)^{m+1}$ signs (Appendix~\ref{app:TETM}).

Only $m = 0, 1, \ldots, M_{\max}$ need to be computed; the negative-$m$
coefficients are obtained by a single sign flip.

\section{FFTLog Hankel Transform}
\label{sec:fftlog}
%=======================================================================

\subsection{Log-variable substitution}

The Hankel transform $A(k_r) = \int_0^\infty f(r)\,J_\nu(k_r r)\,dr$
is converted to a convolution by the substitution $r = r_0\,e^t$,
$k_r = k_0\,e^s$, $r_0 k_0 = 1$:
\begin{align}
  A(e^s/r_0)
  &= \int_{-\infty}^{\infty} f(r_0 e^t)\,J_\nu(e^{s+t})\,r_0\,e^t\,dt
  = \int_{-\infty}^{\infty} g(t)\,K_\nu(s+t)\,dt,
\end{align}
where $g(t) = r_0\,e^t\,f(r_0 e^t)$ and $K_\nu(\tau) = J_\nu(e^\tau)$.
This is a convolution $A = g * K_\nu$, so in Fourier space:
\begin{equation}
  \hat A(q) = \hat g(q)\cdot\hat K_\nu(q).
\end{equation}

\subsection{The filter kernel}

The Mellin transform of $J_\nu(x)$ is
$\mathcal{M}[J_\nu](s) = 2^{s-1}\,\Gamma\!\bigl(\tfrac{\nu+s}{2}\bigr)
\big/\Gamma\!\bigl(\tfrac{\nu+2-s}{2}\bigr)$.
With $x = e^\tau$ and the Mellin variable $s = 1 + iq$ (corresponding to the
kernel $K(t) = t\,J_\nu(t)$ used in the d$\ln r$ measure):
\begin{equation}\label{eq:kernel}
\boxed{
  U_\nu(q)
  = \frac{\Gamma\!\bigl(\frac{\nu+1+iq}{2}\bigr)}
         {\Gamma\!\bigl(\frac{\nu+1-iq}{2}\bigr)}\,2^{+iq}.
}
\end{equation}
The FFTLog computes $\hat A[n] = \hat g[n] \cdot U_\nu(q_n)$, where $\hat g$
is the DFT (FFTW convention $e^{-2\pi i mn/N}$) and $q_n = 2\pi n/(N\Delta\!\ln)$.
The numerator has $+iq$ because the DFT--Mellin product performs a cross-correlation
(argument $s+t$), not a convolution ($s-t$); the Mellin variable $s = +iq$ produces
the correlation kernel.

\textbf{Key property:} the two Gamma arguments are complex conjugates
(both have real part $(\nu+1)/2$ and imaginary parts $\pm q/2$), so
$|U_\nu(q)| = 1$ for all $q$.  There is \emph{no overflow} regardless
of grid size or Bessel order --- in contrast to formulations that use real $q$
in the Gamma arguments, which grow as $\sim 2^{|q|}$ and overflow for
$|q| \gtrsim 300$.

The implementation computes~\eqref{eq:kernel} via
\begin{equation}
  U_\nu(q) = \exp\bigl[
    \mathrm{loggamma}\!\bigl(\tfrac{\nu+1}{2} + \tfrac{iq}{2}\bigr)
    - \mathrm{loggamma}\!\bigl(\tfrac{\nu+1}{2} - \tfrac{iq}{2}\bigr)
    + iq\log 2
  \bigr],
\end{equation}
which avoids overflow at large $|\nu|$.

\paragraph{Output convention.}
The FFTLog output must be \emph{reversed} (the ``natural'' output grid starts
at $1/r_{\min}$; reversing maps it to the useful grid starting at $1/r_{\max}$),
and the result includes an extra factor of $k_r$ from the kernel
$K(t) = t\,J_\nu(t)$.  The caller divides by $k_r$ to obtain the pure Hankel
transform.

\paragraph{Convention warning.}
The Fourier transform of $J_\nu(e^\tau)$ with the $e^{-iq\tau}$ convention gives
$\bar U_\nu(q) = 2^{-iq}\,\Gamma((\nu+1-iq)/2)/\Gamma((\nu+1+iq)/2)$, which is
the \emph{complex conjugate} of~\eqref{eq:kernel}.  Using $\bar U$ instead of $U$
produces a convolution ($s-t$ argument) rather than the required correlation ($s+t$),
giving the wrong transform.  The sign must match: $+iq$ in the numerator for
FFTW's $e^{-2\pi imn/N}$ convention.

\subsection{Discrete implementation}

On a log-spaced grid $r_j = r_0\,e^{j\,\Delta\!\ln}$, $j=0,\ldots,N-1$:
\begin{enumerate}[nosep]
  \item FFT the sampled input: $\hat g[n] = \mathrm{FFT}\{f_r[j]\}$.
  \item Multiply by the kernel: $\hat A[n] = \hat g[n]\cdot\hat K_\nu(q_n)$,
        where $q_n = \frac{2\pi}{N\,\Delta\!\ln}\,n$ with FFTW-ordered indices.
  \item IFFT: $A[j] = \mathrm{IFFT}\{\hat A[n]\}$.
\end{enumerate}
The output lives on the reciprocal log-grid $k_r^{(j)} = (1/r_{N-1})\,e^{j\,\Delta\!\ln}$,
so $k_r^{\max} = 1/r_{\min}$.

\paragraph{Grid constraint.}
To capture all propagating modes ($k_r \le k$), we need
$k_r^{\max} = 1/r_{\min} \ge k$, i.e.\
\begin{equation}
  \boxed{r_{\min} \le \frac{1}{k} = \frac{\lambda}{2\pi}.}
\end{equation}

\paragraph{Comparison with NUFHT.}
Asymptotic Hankel transform methods (Beckman--O'Neil) decompose $J_\nu(x)$ into
oscillatory and decaying parts, using different quadrature for each.  Near the
turning point $x \approx \nu$, neither asymptotic works; accuracy degrades as
$\sim\nu^{-1/3}$.  FFTLog has no such issue: the kernel~\eqref{eq:kernel} is
well-behaved for all~$\nu$.

\subsection{Limitations of FFTLog and the log-polar grid}
\label{sec:fftlog_limitations}

While FFTLog solves the turning-point problem, the log-spaced grid introduces its
own trade-offs:

\paragraph{1.\ Coarse resolution at the aperture edge.}
The grid spacing at radius $r$ is $\Delta r(r) = r\,\Delta\!\ln$.  At the aperture
edge $r = R$, this is the \emph{coarsest} point in the grid.  Resolving near-field
oscillations of wavelength $\lambda$ at $r = R$ requires
\begin{equation}\label{eq:dln_constraint}
  R\,\Delta\!\ln \le \frac{\lambda}{2}
  \qquad\Longrightarrow\qquad
  \Delta\!\ln \le \frac{\lambda}{2R}.
\end{equation}
Meanwhile, the region $r < \lambda$ (where the near field is typically negligible)
receives thousands of grid points it does not need.

\paragraph{2.\ Fixed coupling between $r$ and $k_r$ grids.}
The output grid $k_r^{(j)} = (1/r_{N-1})\,e^{j\Delta\!\ln}$ is fully determined
by the input grid; the two cannot be chosen independently.  To capture all
propagating modes ($k_r \le k$), one needs $k_r^{\max} = 1/r_{\min} \ge k$,
forcing $r_{\min} \le \lambda/(2\pi)$.

For a 2\,mm aperture at $\lambda = 500$\,nm, this gives a dynamic range
$r_{\max}/r_{\min} \ge 2\,\text{mm}/80\,\text{nm} = 25000$.  Combined
with~\eqref{eq:dln_constraint}, the grid size is
\begin{equation}
  N_r = \frac{\ln(r_{\max}/r_{\min})}{\Delta\!\ln}
  \ge \frac{\ln(25000)}{\lambda/(2R)}
  = \frac{10.1}{1.25\times 10^{-4}}
  \approx 81000.
\end{equation}
In practice, rounding to a power of two gives $N_r = 2^{17} = 131072$.

There is no way to ``zoom in'' on the propagating band $k_r \in [0, k]$ without
also changing $r_{\min}$.  The evanescent portion of the $k_r$ grid (which gets
zeroed in the propagation step) represents wasted computation.

\paragraph{3.\ Boundary wrap-around (circular convolution artifact).}
FFTLog computes a \emph{circular} convolution in log-space.  If the input does
not decay to zero near $r_{\min}$ and $r_{\max}$, energy wraps around from one
boundary to the other.  For a lens field that is nonzero at $r = R = r_{\max}$,
this produces spectral leakage.  Mitigation: extend $r_{\max}$ beyond $R$ (padding
with zeros), at the cost of increasing $N_r$ further.

\paragraph{4.\ Hard aperture edges cause ringing.}
A metalens with a sharp amplitude cutoff at $r = R$ produces a Hankel transform
with slowly decaying oscillations (Gibbs-like phenomenon in the spectral domain).
The log-spaced $k_r$ grid may not resolve these oscillations uniformly.  This is
not specific to FFTLog --- any discrete Hankel method exhibits it --- but the
non-uniform spacing makes it harder to diagnose and control.

\paragraph{5.\ Output PSF on a log-spaced grid.}
The output $\rho$-grid inherits the same log spacing.  This is \emph{favourable}
for Airy-disk analysis (finest resolution near $\rho = 0$ where the PSF peaks),
but inconvenient if uniform-$\rho$ output is needed downstream (e.g.\ for imaging
simulation or MTF computation).  Post-hoc interpolation adds a small additional
error.

\paragraph{6.\ No independent zero-padding.}
With standard FFTs one can zero-pad to increase conjugate-domain resolution
without modifying the input grid.  FFTLog ties the two grids together: the only
way to refine the $k_r$ grid is to change $r_{\min}$, $r_{\max}$, or $N_r$.

\paragraph{7.\ Auto-differentiation.}
For metalens inverse design, gradients of the PSF with respect to the lens
transmission $t(r)$ are needed.  FFTLog involves FFTW, complex loggamma, and
non-standard convolution, which most automatic differentiation frameworks cannot
differentiate through without custom adjoint rules.

\paragraph{Assessment.}
All of the above are ``pay more grid points'' problems, not fundamental accuracy
barriers.  By contrast, the NUFHT turning-point issue is a mathematical accuracy
problem that worsens with Bessel order and cannot be resolved simply by adding
grid points.  For the applications considered here (mm-scale lenses at visible
wavelengths), the FFTLog grid overhead is acceptable.

\section{Propagation to the Focal Plane}
\label{sec:propagation}
%=======================================================================

At $z = f$, each eigenmode acquires the phase $e^{ik_z z}$:
\begin{equation}\label{eq:prop}
  \tilde a_m(k_r)
  = a_m(k_r)\,e^{ik_z f},
  \qquad k_z = \sqrt{k^2 - k_r^2}.
\end{equation}
Evanescent modes ($k_r > k \Rightarrow k_z$ imaginary) are set to zero.

The negative-$m$ coefficients are reconstructed via the scalar symmetry
relation~\eqref{eq:scalar_sym}: $\tilde a_{-m} = (-1)^m\,\tilde a_m$.


%=======================================================================
\section{Graf's Addition Theorem: Full Derivation}
\label{sec:graf}
%=======================================================================

\subsection{Statement of the theorem}

Define $x_0 = f\tan\alpha$ and local coordinates
$x = x_0+\rho\cos\psi$, $y = \rho\sin\psi$.
Graf's addition theorem (DLMF \S10.23.7) states that for $\rho < x_0$:
\begin{equation}\label{eq:graf}
  J_n(k_r r)\,e^{in\theta}
  = \sum_{m=-\infty}^{\infty} J_m(k_r x_0)\,J_{n-m}(k_r\rho)\,e^{i(n-m)\psi}.
\end{equation}

\subsection{Substitution into the far-field sum}

The far field at $z=f$ in the global frame is
\begin{equation}
  u(r,\theta,f) = \sum_n\Bigl[\int_0^k \tilde a_n(k_r)\,J_n(k_r r)\,k_r\,dk_r\Bigr]e^{in\theta}.
\end{equation}
Insert~\eqref{eq:graf}:
\begin{align}
  u(\rho,\psi,f)
  &= \sum_n\int_0^k \tilde a_n(k_r)\sum_m J_m(k_r x_0)\,J_{n-m}(k_r\rho)\,e^{i(n-m)\psi}\,k_r\,dk_r.
\end{align}
Relabel $l = n - m$ (so $n = m + l$):
\begin{align}
  u(\rho,\psi,f)
  &= \sum_l\Bigl[\int_0^k
    \underbrace{\sum_m \tilde a_{m+l}(k_r)\,J_m(k_r x_0)}_{B_l(k_r)}
    \,J_l(k_r\rho)\,k_r\,dk_r\Bigr]\,e^{il\psi}.
\end{align}

\begin{tcolorbox}[title=Local modal coefficients (Graf shift)]
\begin{equation}\label{eq:Bl}
  B_l(k_r) = \sum_{|m|\le M_{\max}} \tilde a_{m+l}(k_r)\,J_m(k_r x_0),
  \qquad |l| \le L_{\max}.
\end{equation}
\end{tcolorbox}

\paragraph{Truncation.}
$J_m(k_r x_0)$ decays exponentially for $|m| > k_r x_0$, so the effective
sum length is $\min(M_{\max},\,\lceil k_r x_0\rceil + 20)$.

$L_{\max} \approx k\cdot\mathrm{NA}\cdot\rho_{\max}$: for NA${}=0.5$,
$\rho_{\max}=5\lambda$, $L_{\max}\approx 16 \ll M_{\max}$.

\paragraph{Normal incidence ($\alpha\to 0$).}
$x_0\to 0$, $J_m(0) = \delta_{m,0}$, so $B_l = \tilde a_l$ (identity).


%=======================================================================
\section{Inverse Hankel and Angular Synthesis}
%=======================================================================

\subsection{Inverse Hankel transform (Step~5)}

\begin{equation}\label{eq:bl}
  b_l(\rho) = \int_0^\infty B_l(k_r)\,J_l(k_r\rho)\,k_r\,dk_r
  = \HT_l[k_r\,B_l](\rho).
\end{equation}
This is another FFTLog call (order~$l$) with the integrand pre-multiplied by $k_r$.
Only $2L_{\max}+1$ calls needed.

\subsection{Angular synthesis (Step~6)}

\begin{equation}\label{eq:psf}
  u(\rho_j,\psi_s,f) = \sum_{l=-L_{\max}}^{L_{\max}} b_l(\rho_j)\,e^{il\psi_s},
\end{equation}
evaluated by IFFT over $l$ at each $\rho_j$.  Take $|u|^2$ for PSF intensity.


%=======================================================================
\section{Neumann Shift Theorem: Full Derivation}
\label{sec:neumann}
%=======================================================================

For LPA near fields of the form
$t(r)\cdot e^{ik_x r\cos\theta}$ (axisymmetric transmission $\times$ oblique tilt),
we derive a fast path that avoids $M_{\max}$ individual FFTLog calls.

\subsection{Jacobi--Anger expansion of the tilt}

The identity (DLMF \S10.12.1):
\begin{equation}\label{eq:JA}
  e^{iz\cos\theta} = \sum_{m=-\infty}^{\infty} i^m\,J_m(z)\,e^{im\theta}.
\end{equation}

Applied to the oblique tilt with $z = k_x r$, $k_x = k\sin\alpha$:
\begin{equation}
  e^{ik_x r\cos\theta} = \sum_m i^m\,J_m(k_x r)\,e^{im\theta}.
\end{equation}

\subsection{Scalar modal coefficient}

The $m$-th mode of $u(r,\theta) = t(r)\,e^{ik_x r\cos\theta}$ is
$u_m(r) = i^m\,t(r)\,J_m(k_x r)$
(absorbing $2\pi$ into normalization conventions).  The scalar spectral coefficient is
\begin{equation}\label{eq:am}
  a_m(k_r) = \int_0^\infty u_m(r)\,J_m(k_r r)\,r\,dr
  = i^m\int_0^\infty t(r)\,J_m(k_x r)\,J_m(k_r r)\,r\,dr.
\end{equation}

\subsection{Neumann addition formula: derivation}

We derive the product identity from the Jacobi--Anger expansion.

Start with~\eqref{eq:JA} applied to $e^{ia r\cos\varphi}$:
\begin{equation}
  e^{ia r\cos\varphi} = \sum_n i^n\,J_n(ar)\,e^{in\varphi}.
\end{equation}
Multiply both sides by $e^{-im\varphi}$ and integrate over $\varphi\in[0,2\pi]$:
\begin{equation}\label{eq:JA_proj}
  \int_0^{2\pi} e^{iar\cos\varphi}\,e^{-im\varphi}\,d\varphi
  = 2\pi\,i^m\,J_m(ar).
\end{equation}
(This is Bessel's first integral.)

Now consider the product $J_m(ar)\,J_m(br)$.  Using~\eqref{eq:JA_proj}:
\begin{equation}
  J_m(ar) = \frac{1}{2\pi i^m}\int_0^{2\pi} e^{iar\cos\alpha'}\,e^{-im\alpha'}\,d\alpha'.
\end{equation}
Multiply by $J_m(br) = \frac{1}{2\pi i^m}\int_0^{2\pi} e^{ibr\cos\beta'}\,e^{-im\beta'}\,d\beta'$:
\begin{equation}
  J_m(ar)\,J_m(br) = \frac{1}{(2\pi)^2\,i^{2m}}
  \int_0^{2\pi}\!\!\int_0^{2\pi}
  e^{ir(a\cos\alpha'+b\cos\beta')}\,e^{-im(\alpha'+\beta')}\,d\alpha'\,d\beta'.
\end{equation}
Substitute $\varphi = \alpha'-\beta'$, $\sigma = \alpha'+\beta'$
($d\alpha'\,d\beta' = \tfrac{1}{2}\,d\varphi\,d\sigma$).
Writing $\alpha' = (\sigma+\varphi)/2$, $\beta' = (\sigma-\varphi)/2$:
\begin{equation}
  a\cos\alpha' + b\cos\beta'
  = (a+b)\cos\tfrac{\sigma}{2}\cos\tfrac{\varphi}{2}
  - (a-b)\sin\tfrac{\sigma}{2}\sin\tfrac{\varphi}{2}
  \equiv c(\varphi)\cos(\sigma/2 - \gamma),
\end{equation}
where $c(\varphi) = \sqrt{a^2 + b^2 - 2ab\cos\varphi}$ (by the law of cosines)
and $\gamma$ depends on $\varphi$ but not on $\sigma$.
The $\sigma$-integral then becomes
$\int_0^{4\pi} e^{ic(\varphi)r\cos(\sigma/2-\gamma)}\,e^{-im\sigma}\,d\sigma
= 4\pi\,i^{2m}\,J_{2m}(\ldots)$, which after simplification reduces to
$2\pi\,J_0(c(\varphi)\,r)$ (via the addition theorem for $J_0$).
The net result is:

\begin{tcolorbox}[title=Neumann addition formula]
\begin{equation}\label{eq:neumann}
  J_m(ar)\,J_m(br)
  = \frac{1}{2\pi}\int_0^{2\pi}
    J_0\!\bigl(c(\varphi)\,r\bigr)\,e^{-im\varphi}\,d\varphi,
  \quad c(\varphi) = \sqrt{a^2+b^2-2ab\cos\varphi}.
\end{equation}
\end{tcolorbox}

\subsection{Application to the modal coefficient}

Substitute~\eqref{eq:neumann} into~\eqref{eq:am}:
\begin{align}
  a_m(k_r) &= i^m\int_0^\infty t(r)\,
  \Bigl[\frac{1}{2\pi}\int_0^{2\pi} J_0(c(\varphi)\,r)\,e^{-im\varphi}\,d\varphi\Bigr]\,r\,dr
  \nonumber\\[4pt]
  &= \frac{i^m}{2\pi}\int_0^{2\pi}\Bigl[
    \underbrace{\int_0^\infty r\,t(r)\,J_0(c(\varphi)\,r)\,dr}_{T_0(c(\varphi))}
  \Bigr]\,e^{-im\varphi}\,d\varphi,
\end{align}
where we have exchanged the order of integration and identified the order-0
Hankel transform:
\begin{equation}
  T_0(q) \equiv \HT_0[r\,t(r)](q) = \int_0^\infty r\,t(r)\,J_0(qr)\,dr.
\end{equation}

\begin{tcolorbox}[title=Neumann shift theorem (final)]
\begin{equation}\label{eq:neumann_shift}
  a_m(k_r) = \frac{i^m}{2\pi}\int_0^{2\pi}
    T_0\!\bigl(\sqrt{k_r^2+k_x^2-2k_r k_x\cos\varphi}\,\bigr)\;
    e^{-im\varphi}\,d\varphi.
\end{equation}
\end{tcolorbox}

\noindent
$T_0$ requires a \textbf{single order-0 FFTLog call}.  All $M_{\max}$ modes are
then obtained by evaluating $T_0$ at shifted arguments (interpolation) and
FFT over $\varphi$.

\paragraph{Cost.}
$O(N_r\log N_r)$ for $T_0$, plus $O(N_{k_r}\cdot N_\varphi\log N_\varphi)$ for
the per-$k_r$ interpolation and FFTs.  This replaces $O(M_{\max}\cdot N_r\log N_r)$
FFTLog calls: a factor of $\sim M_{\max}$ improvement.

\paragraph{Practical limitation.}
The interpolation of $T_0$ at the shifted arguments $c(\varphi)$ uses linear
interpolation on the log-spaced $k_r$ grid.  For large $M_{\max}$ (equivalently
large $k_x R$), the arguments $c(\varphi)$ shift far from the evaluation point $k_r$,
crossing regions where $T_0$ oscillates rapidly.  Linear interpolation misses these
oscillations, introducing errors that grow with $M_{\max}$.

In practice, the Neumann path achieves ${<}1\%$ PSF accuracy for
$M_{\max} \lesssim 1600$ (tested at $R \le 500\lambda$, $\alpha \le 30^\circ$).
At production scale ($M_{\max} \approx 12{,}600$, $R = 2\,\text{mm}$,
$\alpha = 30^\circ$), the PSF error reaches ${\sim}9\%$ and the method offers
no speed advantage over the standard per-mode FFTLog path.
Higher-order interpolation (cubic spline) or adaptive oversampling of $T_0$
could extend the useful range, but for the target application the standard
path is recommended.

\section{Ideal Oblique Lens and the Airy Disk}
\label{sec:airy}
%=======================================================================

\subsection{Ideal oblique lens phase}

A lens that focuses perfectly at $(x_0,0,f) = (f\tan\alpha,0,f)$ must produce a
converging spherical wave.  The required phase at the aperture plane is
\begin{equation}\label{eq:oblique_lens}
  u(r,\theta) = \exp\!\Bigl(-ik\Bigl[
    \sqrt{(r\cos\theta - f\tan\alpha)^2 + r^2\sin^2\theta + f^2}
    - f\Bigr]\Bigr).
\end{equation}
The argument of the square root is the distance from aperture point
$(r\cos\theta, r\sin\theta, 0)$ to focal point $(f\tan\alpha, 0, f)$.

This should \emph{not} be confused with
$t(r)\cdot e^{ik\sin\alpha\cdot r\cos\theta}$ (normal lens + oblique plane wave),
which produces coma and is \emph{not} an ideal oblique focus.

\subsection{Effective NA: geometric derivation}

The aperture is a disk of radius $R$ centered at the origin.  The focal spot is at
$(x_0, 0, f)$, which is at distance
\begin{equation}
  d_0 = \sqrt{x_0^2 + f^2} = \sqrt{f^2\tan^2\alpha + f^2} = \frac{f}{\cos\alpha}
\end{equation}
from the aperture center.  The half-angle subtended by the aperture from the focal
spot (in the sagittal plane where the geometry is simplest) is
\begin{equation}
  \sin\theta_{\max} = \frac{R}{\sqrt{R^2 + d_0^2}}
  = \frac{R}{\sqrt{R^2 + f^2/\cos^2\alpha}}
  = \frac{R\cos\alpha}{\sqrt{R^2\cos^2\alpha + f^2}}.
\end{equation}
The effective NA is:
\begin{equation}\label{eq:NA_eff}
  \boxed{
  \mathrm{NA}_{\mathrm{eff}}
  = \frac{R\cos\alpha}{\sqrt{R^2\cos^2\alpha + f^2}}.
  }
\end{equation}
At $\alpha=0$: $\mathrm{NA}_0 = R/\sqrt{R^2+f^2}$.  For $\alpha > 0$:
$\mathrm{NA}_{\mathrm{eff}} < \mathrm{NA}_0$ because $d_0 = f/\cos\alpha > f$.

The Airy disk first zero is at
$\rho_1 = 3.8317/(k\cdot\mathrm{NA}_{\mathrm{eff}}) = 0.6098\,\lambda/\mathrm{NA}_{\mathrm{eff}}$.

\paragraph{Elliptical PSF at large oblique angles.}
The formula~\eqref{eq:NA_eff} is derived in the \emph{sagittal} plane (the
$y$-direction, perpendicular to the tilt).  In this direction, the aperture
appears circular from the focal point and the PSF matches the Airy pattern to
within ${\sim}1$--$2\%$ at all tested angles up to $\alpha = 30^\circ$.

In the \emph{tangential} plane ($\psi = 0$, the tilt direction), the aperture
is foreshortened: the near edge (at $x = -R$) is farther from the focal point
than the far edge (at $x = +R$), producing an effectively asymmetric aperture.
This broadens the PSF in the tangential direction.  The first dark ring shifts
outward by ${\sim}5\%$ at $\alpha = 15^\circ$ and ${\sim}28\%$ at
$\alpha = 30^\circ$.  This is correct diffraction physics for a circular
aperture viewed off-axis, not a numerical artifact.

The simple $\mathrm{NA}_{\mathrm{eff}}$ formula should therefore be used as a
\emph{sagittal-plane} prediction.  For the full 2D PSF shape at large oblique
angles, the pipeline output (which is exact) should be used directly.

\section{Complete Algorithm Summary}
%=======================================================================

\begin{tcolorbox}[title=CyFFP: Scalar Standard Path]
\textbf{Inputs:} $u(r,\theta)$ on log-spaced $r$, uniform $\theta$;\;
$k$, $\alpha$, $f$.

\medskip
\textbf{Step 1} --- Angular FFT: $u_m(r)$ for $m = 0,\ldots,M_{\max}$.

\textbf{Step 2} --- Forward Hankel (FFTLog, order $m$):
$a_m(k_r) = \HT_m[r\,u_m](k_r)$.  One transform per mode.

\textbf{Step 3} --- Propagation + symmetry:
$\tilde a_m = a_m\,e^{ik_z f}$;\;
$\tilde a_{-m} = (-1)^m\,\tilde a_m$ (scalar symmetry).

\textbf{Step 4} --- Graf shift: $B_l = \sum_m \tilde a_{m+l}\,J_m(k_r x_0)$.

\textbf{Step 5} --- Inverse Hankel: $b_l(\rho) = \HT_l[k_r B_l](\rho)$.

\textbf{Step 6} --- Angular IFFT: $u(\rho,\psi) = \sum_l b_l\,e^{il\psi}$.
\end{tcolorbox}

\begin{tcolorbox}[title=CyFFP: Neumann Fast Path (LPA/scalar)]
\textbf{Step A} --- One FFTLog: $T_0 = \HT_0[r\,t(r)]$.

\textbf{Step B} --- Per $k_r$: $\tilde a_m = i^m\,\mathrm{FFT}_\varphi\{T_0(c(\varphi))\}[m]\cdot e^{ik_z f}$.

\textbf{Steps 4--6} --- Unchanged.
\end{tcolorbox}


%=======================================================================
\section{LPA Metalens Driver (\texttt{compute\_psf})}
\label{sec:compute_psf}
%=======================================================================

The function \texttt{compute\_psf} provides a user-facing interface for
computing the PSF of a metalens described by its radial transmission
$t(r)$ in the local phase approximation (LPA).

\subsection{Input model}

The metalens is divided into $N_{\mathrm{cells}}$ concentric annular unit cells
of radial width (pitch) $\Lambda$.  Each cell~$i$ has a complex transmission
coefficient $t_i$, representing the transmitted near field averaged over the
cell.  The field is assumed piecewise constant:
\begin{equation}
  t(r) = t_i \quad\text{for}\quad (i{-}1)\Lambda < r \le i\Lambda,
  \qquad i = 1,\ldots,N_{\mathrm{cells}}.
\end{equation}
The aperture radius is $R = N_{\mathrm{cells}}\,\Lambda$.
Under oblique illumination at angle~$\alpha$, the LPA near field is
\begin{equation}
  u(r,\theta) = t(r)\,e^{ik_x r\cos\theta},
  \qquad k_x = k\sin\alpha.
\end{equation}

\subsection{Mode construction via Jacobi--Anger}

The Jacobi--Anger expansion gives the modes analytically:
\begin{equation}\label{eq:um_jacobi}
  u_m(r) = i^m\,t(r)\,J_m(k_x r),
  \qquad m = 0,\ldots,M_{\max}.
\end{equation}
This replaces Step~1 (angular FFT) entirely.  Computing all $M_{\max}+1$
Bessel values at each $r$ is done in a single Miller backward recurrence
sweep, giving $J_0(k_x r),\ldots,J_{M_{\max}}(k_x r)$ in $O(M_{\max})$
time per radial point.  The total cost is $O(N_r \cdot M_{\max})$, which
is negligible compared to Step~2.

\paragraph{Resampling from uniform to log grid.}
The transmission $t(r)$ is provided on a uniform grid (one value per cell).
Since $t$ is piecewise constant, resampling to the FFTLog log-spaced grid is
a simple cell lookup: $t(r_j) = t_{\lceil r_j/\Lambda\rceil}$.  This is exact
for the LPA model.

\subsection{Pipeline}

After mode construction, the standard pipeline runs:
\begin{center}
\begin{tabular}{@{}r p{10cm}@{}}
\textbf{Step 2} & Scalar FFTLog: $a_m(k_r) = \HT_m[r\,u_m](k_r)$, one call per mode. \\
\textbf{Step 3} & Propagation: $\tilde a_m = a_m\,e^{ik_z f}$;\;
                   $\tilde a_{-m} = (-1)^m\tilde a_m$. \\
\textbf{Step 4} & Graf shift: $B_l = \sum_m \tilde a_{m+l}\,J_m(k_r x_0)$. \\
\textbf{Step 5} & Inverse Hankel: $b_l(\rho) = \HT_l[k_r B_l](\rho)$. \\
\textbf{Step 6} & Angular synthesis: $u(\rho,\psi) = \sum_l b_l\,e^{il\psi}$.
\end{tabular}
\end{center}

\subsection{Output}

The PSF is interpolated from the native $(\rho,\psi)$ log-polar grid onto
a uniform Cartesian $(x,y)$ grid centered at the focal spot, using bilinear
interpolation in $(\ln\rho,\,\psi)$.  The output intensity $|u(x,y)|^2$ is
returned both normalized (peak = 1) and unnormalized, along with the axes
in physical units ($\mu$m) and in units of $\lambda$.

\subsection{Timing estimate (2\,mm lens, $\alpha=30^\circ$, 388 threads)}

\begin{center}
\begin{tabular}{@{}l r l@{}}
\toprule
\textbf{Stage} & \textbf{Time} & \textbf{Notes} \\
\midrule
Mode construction  & ${\sim}3$\,s  & Jacobi--Anger + Miller recurrence \\
Step 2 (FFTLog)    & ${\sim}50$\,s & 12\,588 modes $\times$ Nr=131\,072 \\
Step 3 (propagate) & ${\sim}15$\,s & Memory-bandwidth limited \\
Step 4 (Graf)      & ${\sim}5$\,s  & Miller + Bessel truncation \\
Steps 5--6         & ${\sim}1$\,s  & Only $2L_{\max}+1 \approx 31$ calls \\
\midrule
\textbf{Total}     & ${\sim}\mathbf{74}$\,s & vs 303\,s with per-$r$ $\theta$-FFT \\
\bottomrule
\end{tabular}
\end{center}

\noindent
The $4\times$ speedup over the per-$r$ $\theta$-FFT approach comes from replacing
the $O(N_r \cdot N_\theta \log N_\theta)$ angular FFT with the
$O(N_r \cdot M_{\max})$ Jacobi--Anger recurrence.

\subsection{Plan/execute pattern for optimization loops}

For inverse design, the PSF is evaluated many times with different
$t(r)$ but fixed geometry ($R$, $\lambda$, $\alpha$, $f$).  The function
\texttt{prepare\_psf} precomputes all geometry-dependent data:

\begin{itemize}[nosep]
  \item $J_m(k_x r_j)$ for all $(m,j)$: the Jacobi--Anger Bessel matrix
        ($N_r \times M_{\max}$ reals, ${\sim}13$\,GB for 2\,mm lens).
        Uses Miller backward recurrence, threaded over~$r$.
  \item FFTLog filter kernels $U_m(q)$ for all~$m$: precomputed via the
        $\Gamma$-recurrence ($N_r \times M_{\max}$ complex, ${\sim}26$\,GB).
  \item FFTW in-place plans (one per thread, created with \texttt{MEASURE}).
  \item Cell lookup table: maps each log-grid index to its unit cell.
\end{itemize}
The total precomputation cost is ${\sim}15$\,s and ${\sim}40$\,GB for the
2\,mm production case.  The function \texttt{execute\_psf} then takes a new
$t(r)$ vector and runs only:
\begin{enumerate}[nosep]
  \item Resample $t$ via cell lookup (trivial).
  \item $u_m = i^m\,t \cdot J_m$ (element-wise multiply with precomputed matrix).
  \item FFTLog with precomputed kernels and plans (${\sim}50$\,s, irreducible).
  \item Steps 3--6 (${\sim}20$\,s).
\end{enumerate}
Per-iteration cost: ${\sim}55$\,s (vs ${\sim}74$\,s for \texttt{compute\_psf}).
Over 100 design iterations, the precomputation saves ${\sim}30$ minutes.


%=======================================================================
\section{Comparison with the Brute-Force Cartesian Approach}
\label{sec:cartesian}
%=======================================================================

The most straightforward alternative is to place the near field on a 2D Cartesian
grid and propagate via 2D FFT (angular spectrum method):
\begin{enumerate}[nosep]
  \item Sample $u(x,y,0)$ on a uniform grid with spacing $\Delta x \le \lambda/2$
        covering $[-R, R]^2$.
  \item 2D FFT $\to$ angular spectrum $U(k_x, k_y)$.
  \item Multiply by $e^{ik_z f}$, zero out evanescent modes.
  \item 2D IFFT $\to$ field at $z = f$.  Extract PSF near $(f\tan\alpha, 0)$.
\end{enumerate}

\paragraph{Grid size.}
$N_x = N_y = 2R/(\lambda/2) = 4R/\lambda$.
For $R = 2$\,mm, $\lambda = 500$\,nm: $N_x = 16000$.  The grid has
${\sim}2.6\times 10^8$ points.

\paragraph{Cost and memory.}
Each complex array occupies $N_x^2 \times 16$\,bytes $\approx 4.3$\,GB.
Two to three working arrays require ${\sim}10$--$13$\,GB.
The two 2D FFTs cost $O(N_x^2 \log N_x^2)$ $\approx 7.5\times 10^9$ operations,
taking ${\sim}3$--$4$ seconds on a modern CPU.

\paragraph{Scaling.}
The critical weakness is that the grid grows as $(R/\lambda)^2$:

\begin{center}
\begin{tabular}{@{}lccc@{}}
\toprule
$R/\lambda$ & Cartesian grid & Memory (per array) & CyFFP $N_r$ (1D) \\
\midrule
4\,000 (2\,mm, 500\,nm) & $16\,384^2$ & 4.3\,GB & 131\,072 \\
10\,000 (5\,mm)          & $40\,960^2$ & 27\,GB  & 327\,680 \\
20\,000 (1\,cm)          & $81\,920^2$ & 107\,GB & 655\,360 \\
\bottomrule
\end{tabular}
\end{center}

\noindent
For a 1\,cm visible-wavelength lens the Cartesian approach requires ${\sim}100$\,GB
per array and is impractical on commodity hardware, while CyFFP uses a 1D radial
grid of ${\sim}655$\,K points (${\sim}10$\,MB).

\paragraph{When Cartesian wins.}
If the full field over the entire focal plane is needed (not just the PSF patch),
or if the structure is \emph{not} axisymmetric (so cylindrical decomposition
does not apply), the Cartesian approach is the only option.

\section{Implementation Recommendations}
\label{sec:implementation}
%=======================================================================

\subsection{Recommended optimizations}

An efficient implementation should incorporate the following:

\begin{center}
\begin{tabular}{@{}l l p{7cm}@{}}
\toprule
\textbf{Optimization} & \textbf{Speedup} & \textbf{Mechanism} \\
\midrule
Negative-$m$ symmetry
  & $2\times$ on Step~2
  & Compute only $m \ge 0$; reconstruct $m < 0$ by
    $\tilde a_{-m} = (-1)^m\tilde a_m$ (scalar). \\[6pt]
Neumann shift (LPA fields)
  & ${\sim}M_{\max}\times$ on Step~2
  & For fields of the form $t(r)\,e^{ik_x r\cos\theta}$, replace
    all $M_{\max}$ per-mode FFTLog calls by a single order-0 call
    followed by interpolation and FFT at each $k_r$
    (\S\ref{sec:neumann}). \\[6pt]
Bessel truncation in Graf shift
  & ${\sim}4\times$ on Step~4
  & $J_m(k_r x_0)$ decays exponentially for $|m| > k_r x_0$.
    Truncate the inner sum at
    $|m| \le \min(M_{\max},\,\lceil k_r x_0\rceil + 20)$. \\[6pt]
Miller backward recurrence
  & ${\sim}10\times$ on Bessel eval
  & Compute $J_0(x), \ldots, J_{M}(x)$ via the recurrence
    $J_{m-1}(x) = (2m/x)\,J_m(x) - J_{m+1}(x)$,
    starting from $J_{m_{\mathrm{start}}} = 1$, $J_{m_{\mathrm{start}}+1} = 0$
    and normalising via the self-normalizing identity
    $1 = J_0(x) + 2\sum_{k\ge 1} J_{2k}(x)$.
    The starting index must satisfy
    $m_{\mathrm{start}} \ge \max(M, \lceil|x|\rceil) + \max(30,\, 15\sqrt{|x|})$.
    \textbf{Caution:} the coefficient is $2m/x$, not $2(m{+}1)/x$;
    overflow protection (periodic rescaling) is needed for large~$x$. \\[6pt]
SIMD vectorization
  & $2$--$4\times$ on Step~4
  & The inner loop $\sum_m \tilde a_{m+l}\,J_m$ is a dot product
    over contiguous memory and should be marked for SIMD
    auto-vectorization. \\[6pt]
Shared-memory threading
  & ${\sim}N_{\mathrm{threads}}\times$
  & Steps 2, 4, 5, and 6 are embarrassingly parallel (over modes,
    $k_r$ points, local modes, and $\rho$ points respectively).
    Shared-memory threads avoid serialization overhead. \\
\bottomrule
\end{tabular}
\end{center}

\subsection{Threading considerations}

Steps 2 and 4 dominate the computation.  Both are embarrassingly parallel:
\begin{itemize}[nosep]
  \item \textbf{Step 2}: each mode $m$ is independent.  Shared-memory threads
        avoid the serialization overhead of distributed workers (the modal data
        matrices can be large: $N_r \times M_{\max} \times 16$\,bytes).
  \item \textbf{Step 4}: each $k_r$ point is independent.
  \item \textbf{Steps 5--6}: each local mode $l$ (Step~5) or radial point $\rho_j$
        (Step~6) is independent.
\end{itemize}
Care is needed with FFT plan caching: the FFT plan for a given array size should
be created once (single-threaded) before entering the parallel region.

\subsection{Entry points}

A well-designed implementation should provide at least three entry points:
\begin{enumerate}[nosep]
  \item \textbf{Full-field driver}: accepts $u(r,\theta)$ on a 2D grid, performs
        the angular FFT (Step~1) internally, then runs Steps 2--6.
        Useful for validation with analytical 2D fields.
  \item \textbf{Modal driver}: accepts pre-decomposed modes $u_m(r)$
        for $m \ge 0$ directly from FDTD/RCWA.
        Skips Step~1 entirely.  This is the primary production interface.
  \item \textbf{Neumann shift driver}: accepts only the radial transmission $t(r)$
        and the oblique angle.  Uses the Neumann shift theorem (\S\ref{sec:neumann})
        to bypass per-mode FFTLog calls.  Applicable when the near field factors as
        $t(r)\cdot e^{ik_x r\cos\theta}$ (LPA / angle-insensitive metalens model).
\end{enumerate}

\appendix
\section{Vector TE/TM Formulation}
\label{app:TETM}
%=======================================================================

This appendix presents the full vector TE/TM decomposition.  It is retained
for completeness and for future extensions to full-vector PSF computation
(both Cartesian components).

\textbf{Important caveat:} In the scalar pipeline (which is the primary
formulation used in CyFFP), the combined coefficient
$\tilde A_m = A^{\mathrm{TE}}_m + A^{\mathrm{TM}}_m$ is \emph{not} used.
Naively combining TE and TM coefficients conflates different Bessel structures
for the $r$- and $\theta$-components and produces incorrect PSF results (e.g.,
the PSF peak shifts away from $\rho = 0$).  The scalar coefficient
$a_m = \HT_m[r\,u_m]$ (\S\ref{sec:scalar_coeffs}) is the correct approach
for a single Cartesian polarization component.

\subsection{Scalar Helmholtz equation in cylindrical coordinates}

In a source-free region, the electric and magnetic fields can be derived from
scalar potentials satisfying the Helmholtz equation
\begin{equation}\label{eq:helmholtz}
  \nabla^2\psi + k^2\psi = 0,\qquad k = \frac{2\pi n}{\lambda} = \omega\sqrt{\mu\epsilon}.
\end{equation}
In cylindrical coordinates:
\begin{equation}
  \frac{1}{r}\frac{\partial}{\partial r}\Bigl(r\frac{\partial\psi}{\partial r}\Bigr)
  + \frac{1}{r^2}\frac{\partial^2\psi}{\partial\theta^2}
  + \frac{\partial^2\psi}{\partial z^2}
  + k^2\psi = 0.
\end{equation}
Separation of variables $\psi = R(r)\,\Theta(\theta)\,Z(z)$ gives:
\begin{itemize}[nosep]
  \item $Z'' = -k_z^2 Z$ \;$\Rightarrow$\; $Z = e^{\pm ik_z z}$,
  \item $\Theta'' = -m^2\Theta$ \;$\Rightarrow$\; $\Theta = e^{im\theta}$,
  \item $r(rR')' + (k_r^2 r^2 - m^2)R = 0$ \;$\Rightarrow$\; $R = J_m(k_r r)$,
\end{itemize}
where $k_r^2 + k_z^2 = k^2$.  The complete solution is
\begin{equation}\label{eq:psi}
  \psi = J_m(k_r r)\,e^{im\theta}\,e^{ik_z z}.
\end{equation}


\subsection{TM harmonics}

Set the vector potential $\A/\mu = \bpsi = (0,0,\psi_z)$ with $\psi_z = \psi$ from~\eqref{eq:psi}.
The tangential electric field (absorbing impedance $\eta$ into the coefficient) is:
\begin{equation}\label{eq:ETM_tang}
\boxed{
  \E^{\mathrm{TM},\perp}_{m,k_r}
  = \eta\Bigl[
    -\frac{k_z}{k}\,\frac{\partial J_m(k_r r)}{\partial r}\,\rhat
    \;-\; i\,\frac{m\,k_z}{k\,r}\,J_m(k_r r)\,\that
  \Bigr]\,e^{im\theta}\,e^{ik_z z}.
}
\end{equation}


\subsection{TE harmonics}

Set $\F/\epsilon = \bpsi = (0,0,\psi_z)$.  Then $\E^{\mathrm{TE}} = -\nabla\times\bpsi$:
\begin{equation}\label{eq:ETE_tang}
\boxed{
  \E^{\mathrm{TE},\perp}_{m,k_r}
  = \Bigl[
    -\frac{im}{r}\,J_m(k_r r)\,\rhat
    + \frac{\partial J_m(k_r r)}{\partial r}\,\that
  \Bigr]\,e^{im\theta}\,e^{ik_z z}.
}
\end{equation}


\subsection{TE and TM expansion coefficients}

The TE coefficient is obtained by projecting the near field onto $\E^{\mathrm{TE}*}$
and using Bessel orthogonality.  Substituting recurrence
relations~\eqref{eq:moverx}--\eqref{eq:dJdr}:

\begin{equation}\label{eq:ATE}
  A^{\mathrm{TE}}_m(k_r)
  = \frac{i}{2}\bigl[\HT_{m+1}[r\,E_{m,r}] + \HT_{m-1}[r\,E_{m,r}]\bigr]
  + \frac{1}{2}\bigl[\HT_{m-1}[r\,E_{m,\theta}] - \HT_{m+1}[r\,E_{m,\theta}]\bigr].
\end{equation}

Similarly:
\begin{equation}\label{eq:ATM}
  A^{\mathrm{TM}}_m(k_r)
  = -\frac{k_z/k}{2}\bigl[\HT_{m-1}[rE_{m,r}] - \HT_{m+1}[rE_{m,r}]\bigr]
  - \frac{ik_z/k}{2}\bigl[\HT_{m-1}[rE_{m,\theta}] + \HT_{m+1}[rE_{m,\theta}]\bigr].
\end{equation}

Each mode requires \textbf{4 Hankel transforms} at orders $m\pm 1$.


\subsection{Negative-$m$ symmetry (TE/TM)}

For $y$-polarized illumination ($E_{-m,r} = -E_{m,r}$, $E_{-m,\theta} = +E_{m,\theta}$):
\begin{equation}\label{eq:sym_ypol}
  A^{\mathrm{TE}}_{-m} = (-1)^{m}\,A^{\mathrm{TE}}_m, \qquad
  A^{\mathrm{TM}}_{-m} = (-1)^{m+1}\,A^{\mathrm{TM}}_m.
\end{equation}

For $x$-polarized illumination ($E_{-m,r} = +E_{m,r}$, $E_{-m,\theta} = -E_{m,\theta}$):
\begin{equation}\label{eq:sym_xpol}
  A^{\mathrm{TE}}_{-m} = (-1)^{m+1}\,A^{\mathrm{TE}}_m, \qquad
  A^{\mathrm{TM}}_{-m} = (-1)^{m}\,A^{\mathrm{TM}}_m.
\end{equation}


%=======================================================================
\section{Adjoint (Reverse-Mode Differentiation) of the PSF Pipeline}
\label{sec:adjoint}
%=======================================================================

For gradient-based design optimization, we require $\partial\mathcal{L}/\partial t_i$,
where $\mathcal{L}(I_{\mathrm{cart}})$ is a real-valued loss function of the PSF
intensity and $t_i$ are the complex-valued transmission coefficients of the metalens
unit cells.  We derive the adjoint (pullback) of each pipeline step.

\subsection{Notation and conventions}

For a real-valued loss $\mathcal{L}$ and complex-valued intermediate variable~$z$,
the \emph{cotangent} (adjoint variable) is
\begin{equation}
  \bar z \;\equiv\; \frac{\partial\mathcal{L}}{\partial z^*}
  \qquad\text{(Wirtinger derivative w.r.t.\ the conjugate)}.
\end{equation}
The total differential satisfies
$d\mathcal{L} = 2\,\mathrm{Re}\bigl[\bar z^*\,dz\bigr]$.
For a real variable~$x$, $\bar x = \partial\mathcal{L}/\partial x$ as usual.

For a linear map $y = Ax$, the pullback is $\bar x = A^H \bar y$.

The FFTW conventions: $\mathrm{FFT}(x)[k] = \sum_j x_j\,e^{-2\pi i jk/N}$,\;
$\mathrm{IFFT}(X)[j] = \tfrac{1}{N}\sum_k X_k\,e^{+2\pi ijk/N}$.
Pullback of $\mathrm{FFT}$: $\bar x = N\cdot\mathrm{IFFT}(\bar y)$.
Pullback of $\mathrm{IFFT}$: $\bar X = \tfrac{1}{N}\cdot\mathrm{FFT}(\bar x)$.


\subsection{Forward pipeline summary}

The forward chain from $t_i$ to $I_{\mathrm{cart}}$ is:
\begin{equation}\label{eq:forward_chain}
  t_i \;\xrightarrow{1}\; t(r_j) \;\xrightarrow{2}\; u_m(r_j)
  \;\xrightarrow{3}\; a_m(k_r) \;\xrightarrow{4}\; \tilde a_m(k_r)
  \;\xrightarrow{5}\; B_l(k_r) \;\xrightarrow{6}\; b_l(\rho)
  \;\xrightarrow{7}\; u(\rho,\psi) \;\xrightarrow{8}\; I(\rho,\psi)
  \;\xrightarrow{9}\; I_{\mathrm{cart}}(x,y).
\end{equation}
All steps except~8 (modulus squared) are linear in the upstream variable.


\subsection{Step 9 adjoint: Cartesian interpolation (scatter)}

Forward (bilinear interpolation, gather):
\begin{equation}
  I_{\mathrm{cart}}[i_y,i_x]
  = \sum_{\alpha\in\{00,10,01,11\}} w_\alpha\;I_{\mathrm{polar}}[j_\alpha, p_\alpha],
\end{equation}
where $w_\alpha$ are the bilinear weights and $(j_\alpha, p_\alpha)$ are the four
bracketing grid indices in $(\ln\rho,\,\psi)$.

Adjoint (scatter-add): for each $(i_y,i_x)$,
\begin{equation}\label{eq:adj_interp}
  \bar I_{\mathrm{polar}}[j_\alpha, p_\alpha]
  \;\mathrel{+}=\; w_\alpha\;\bar I_{\mathrm{cart}}[i_y, i_x].
\end{equation}


\subsection{Step 8 adjoint: intensity $I = |u|^2$}

This is the only nonlinear step.  The Wirtinger derivative
$\partial(u\,u^*)/\partial u^* = u$ gives:
\begin{equation}\label{eq:adj_abs2}
  \bar u(\rho_j,\psi_s) = \bar I_{\mathrm{polar}}[j,s]\;\cdot\;u(\rho_j,\psi_s).
\end{equation}
Note: no factor of~2 (it is already absorbed into the identity
$d\mathcal{L} = 2\,\mathrm{Re}[\bar u^*\,du]$), and no conjugation of~$u$
(the Wirtinger derivative of $|u|^2$ w.r.t.\ $u^*$ is $u$, not $u^*$).

\textbf{Requires the saved forward field $u(\rho,\psi)$.}
This is the only quantity that must be stored from the forward pass
($N_r \times N_\psi$ complex values, ${\sim}130$\,MB at production scale).


\subsection{Step 7 adjoint: angular synthesis}

Forward: $u(\rho_j,\psi_s) = N_\psi\cdot\mathrm{IFFT}_\psi\{b_l(\rho_j)\}[s]$.

Adjoint (pullback of $N_\psi\cdot\mathrm{IFFT}$ is $\mathrm{FFT}$):
\begin{equation}\label{eq:adj_synth}
  \bar b_l(\rho_j) = \mathrm{FFT}_\psi\{\bar u(\rho_j,\psi_s)\}[l]
\end{equation}
with the appropriate FFTW index reordering for negative~$l$.


\subsection{Step 6 adjoint: inverse Hankel transform}

Forward: $b_l(\rho) = \mathrm{FFTLog}_l\bigl(k_r\,B_l(k_r)\bigr)\big/\rho$.

The FFTLog at order~$l$ is the composition:
\begin{equation}
  \mathrm{raw} = \mathrm{Reverse}\bigl[\mathrm{IFFT}\bigl(
    U_l \cdot \mathrm{FFT}(f)\bigr)\bigr],
  \qquad f = k_r\,B_l.
\end{equation}

Adjoint of $b = \mathrm{raw}/\rho$: $\;\overline{\mathrm{raw}} = \bar b / \rho$
\quad(since $\rho$ is real positive).

Adjoint of the FFTLog core (derived by chaining pullbacks of Reverse, IFFT,
$\mathrm{diag}(U_l)$, FFT):
\begin{equation}\label{eq:adj_fftlog}
\boxed{
  \bar f = \mathrm{IFFT}\Bigl(
    U_l^* \cdot \mathrm{FFT}\bigl(\mathrm{Reverse}(\overline{\mathrm{raw}})\bigr)
  \Bigr).
}
\end{equation}
Since $|U_l(q)| = 1$, we have $U_l^* = 1/U_l$: the adjoint FFTLog uses the
\emph{conjugate kernel}, which is the inverse Hankel transform kernel.
The structure is identical to the forward FFTLog, confirming that the
Hankel transform is (formally) self-adjoint.

Adjoint of $f = k_r\,B_l$: $\;\bar B_l = k_r\,\bar f$
\quad(since $k_r$ is real).


\subsection{Step 5 adjoint: Graf shift}

Forward: $B_l(k_r) = \sum_{m} \tilde a_{m+l}(k_r)\,J_m(k_r x_0)$,\;
$|l| \le L_{\max}$,\; $|m| \le m_{\mathrm{cut}}(k_r)$.

Adjoint (transpose the convolution):
\begin{equation}\label{eq:adj_graf}
  \overline{\tilde a}_n(k_r) = \sum_{l=-L_{\max}}^{L_{\max}}
  \bar B_l(k_r)\;J_{n-l}(k_r x_0),
  \qquad |n| \le M_{\max}.
\end{equation}
The Bessel weights $J_{n-l}(k_r x_0)$ are real, so no conjugation is needed.
The same Miller recurrence, Bessel truncation, and $J_{-m} = (-1)^m J_m$
symmetry apply as in the forward.


\subsection{Step 4 adjoint: propagation and symmetry reconstruction}

Forward: $\tilde a_m = a_m\,e^{ik_z f}$ for $m \ge 0$;\;
$\tilde a_{-m} = (-1)^m\,\tilde a_m$ for $m > 0$.

The positive- and negative-$m$ slots of $\tilde a$ both depend on the single
$a_m$ coefficient.  Adjoint:
\begin{equation}\label{eq:adj_prop}
  \bar a_m = \begin{cases}
    e^{-ik_z f}\;\overline{\tilde a}_0 & m = 0,\\[4pt]
    e^{-ik_z f}\;\bigl[\overline{\tilde a}_m + (-1)^m\,\overline{\tilde a}_{-m}\bigr]
    & m = 1,\ldots,M_{\max}.
  \end{cases}
\end{equation}
The $m = 0$ case has no negative counterpart ($\tilde a_0$ occupies a single
slot in the array; the general formula would double-count it).
The factor $e^{-ik_z f} = (e^{ik_z f})^*$ is the conjugate of the
forward propagation phase ($k_z$ is real for propagating modes, and the
evanescent modes have $\tilde a = 0$).


\subsection{Step 3 adjoint: FFTLog (scalar spectral coefficients)}

Forward: $a_m = \mathrm{FFTLog}_m(r\,u_m)\big/k_r$.

Adjoint of $a_m = \mathrm{raw}/k_r$: $\;\overline{\mathrm{raw}} = \bar a_m / k_r$.

Adjoint of the FFTLog (same formula as~\eqref{eq:adj_fftlog} with order~$m$):
\begin{equation}
  \bar g = \mathrm{IFFT}\Bigl(U_m^*\cdot\mathrm{FFT}\bigl(
    \mathrm{Reverse}(\overline{\mathrm{raw}})\bigr)\Bigr).
\end{equation}

Adjoint of $g = r\,u_m$: $\;\bar u_m = r\,\bar g$\quad(since $r$ is real).


\subsection{Step 2 adjoint: mode construction}

Forward: $u_m(r_j) = i^m\;t(r_j)\;J_m(k_x r_j)$.

Since $J_m$ and $t$ are both functions of $r_j$, and $i^m$ is a constant per mode:
\begin{equation}\label{eq:adj_modes}
  \bar t(r_j) = \sum_{m=0}^{M_{\max}} (-i)^m\;J_m(k_x r_j)\;\bar u_m(r_j),
\end{equation}
using $\overline{(i^m)} = (i^m)^* = (-i)^m$.
The $J_m(k_x r_j)$ values are the precomputed Bessel matrix from the plan.


\subsection{Step 1 adjoint: resampling (scatter-add)}

Forward (gather): $t(r_j) = t_{c(j)}$, where $c(j) = \lceil r_j/\Lambda\rceil$
is the cell index.

Adjoint (scatter-add):
\begin{equation}\label{eq:adj_resample}
  \bar t_i = \sum_{\{j\,:\,c(j)=i\}} \bar t(r_j).
\end{equation}


\subsection{Summary: adjoint pipeline}

\begin{tcolorbox}[title=Adjoint of \texttt{execute\_psf}]
\textbf{Given:} $\bar I_{\mathrm{cart}} = \partial\mathcal{L}/\partial I_{\mathrm{cart}}$
and saved forward field $u(\rho,\psi)$.

\medskip
\textbf{9.}\ Scatter: $\bar I_{\mathrm{polar}}$ from $\bar I_{\mathrm{cart}}$
\hfill\eqref{eq:adj_interp}

\textbf{8.}\ $\bar u = \bar I_{\mathrm{polar}}\cdot u$
\hfill\eqref{eq:adj_abs2}

\textbf{7.}\ $\bar b_l = \mathrm{FFT}_\psi(\bar u)[l]$
\hfill\eqref{eq:adj_synth}

\textbf{6.}\ For each $l$: $\bar B_l = k_r \cdot \mathrm{IFFT}(U_l^*\cdot\mathrm{FFT}(\mathrm{Rev}(\bar b_l/\rho)))$
\hfill\eqref{eq:adj_fftlog}

\textbf{5.}\ $\overline{\tilde a}_n = \sum_l \bar B_l\,J_{n-l}(k_r x_0)$
\hfill\eqref{eq:adj_graf}

\textbf{4.}\ $\bar a_m = e^{-ik_z f}[\overline{\tilde a}_m + (-1)^m\overline{\tilde a}_{-m}]$
\hfill\eqref{eq:adj_prop}

\textbf{3.}\ For each $m$: $\bar u_m = r\cdot\mathrm{IFFT}(U_m^*\cdot\mathrm{FFT}(\mathrm{Rev}(\bar a_m/k_r)))$
\hfill\eqref{eq:adj_fftlog}

\textbf{2.}\ $\bar t(r_j) = \sum_m (-i)^m\,J_m(k_x r_j)\,\bar u_m(r_j)$
\hfill\eqref{eq:adj_modes}

\textbf{1.}\ $\bar t_i = \sum_{\{j:c(j)=i\}} \bar t(r_j)$
\hfill\eqref{eq:adj_resample}
\end{tcolorbox}

\noindent
\textbf{Memory:} only $u(\rho,\psi)$ must be saved from the forward pass
(${\sim}130$\,MB).  The adjoint allocates the same large arrays as the forward
($\overline{\tilde a}$: ${\sim}53$\,GB peak).

\noindent
\textbf{Cost:} approximately equal to the forward pass (${\sim}55$\,s), dominated
by $M_{\max}$ adjoint FFTLog calls in Step~3.

\noindent
\textbf{Verification:} finite-difference check
$[\mathcal{L}(t{+}\varepsilon\delta) - \mathcal{L}(t{-}\varepsilon\delta)]/(2\varepsilon)
\approx 2\,\mathrm{Re}[\bar t^\dagger\delta]$
for random complex perturbation~$\delta$, where $\bar t^\dagger\delta = \sum_i \bar t_i^*\,\delta_i$.
Verified to $10^{-10}$ relative error for total intensity, peak pixel,
random-weighted, and single-cell perturbation losses.

\paragraph{Fused Steps 3+2.}
In the implementation, the adjoint FFTLog (Step~3) and mode construction
(Step~2) are fused into a single threaded loop: as each mode's
$\bar u_m$ is computed in a per-thread buffer, it is immediately accumulated
into $\bar t$ via $\bar t(r_j) \mathrel{+}= (-i)^m J_m(k_x r_j)\,\bar u_m(r_j)$.
Per-thread accumulators are reduced after the loop.  This eliminates the
$\bar u_m[N_r \times (M_{\max}{+}1)]$ array (${\sim}26$\,GB at production scale).

\subsection{Usage: differentiable PSF}

\paragraph{Manual gradient computation.}
The function \texttt{psf\_adjoint(plan, t\_vals, result, dL\_dI\_raw)} computes
$\partial\mathcal{L}/\partial t$ given $\partial\mathcal{L}/\partial I_{\mathrm{raw}}$:
\begin{verbatim}
  plan   = prepare_psf(pitch, N_cells; ...)   # precompute (once)
  result = execute_psf(plan, t_vals)           # forward (~55s)
  dL_dI  = your_loss_gradient(result.I_raw)    # Matrix{Float64}
  dL_dt  = psf_adjoint(plan, t_vals, result, dL_dI)  # adjoint (~2-4× fwd)
\end{verbatim}

\paragraph{Zygote automatic differentiation.}
The function \texttt{psf\_intensity(plan, t\_vals)} returns only the
unnormalised PSF intensity $I_{\mathrm{raw}}$ as a plain \texttt{Matrix\{Float64\}}.
A \texttt{ChainRulesCore.rrule} is defined in the separate file
\texttt{cyffp\_zygote.jl}, which intercepts \texttt{psf\_intensity} and routes
the adjoint through \texttt{psf\_adjoint}.

\begin{verbatim}
  include("cyffp.jl");  using .CyFFP
  using ChainRulesCore, Zygote
  include("cyffp_zygote.jl")

  plan = prepare_psf(0.3, 6667; lambda_um=0.5, alpha_deg=30.0, NA=0.4)
  grad = Zygote.gradient(t -> begin
      I = psf_intensity(plan, t)
      return -I[cy, cx] / (sum(I) + 1e-10)   # Strehl-like loss
  end, t_vals)[1]
\end{verbatim}

\noindent
The \texttt{rrule} is defined for \texttt{psf\_intensity} (not
\texttt{execute\_psf}) because the plain matrix return type avoids fragile
\texttt{NamedTuple} cotangent extraction.  Zygote automatically differentiates
the loss function composition (indexing, arithmetic, etc.)\ outside the PSF
computation.

\paragraph{Scaling.}
The adjoint/forward cost ratio follows ${\sim}0.7\,M_{\max}^{0.17}$, plateauing
at ${\sim}2.4\times$ for $M_{\max} > 200$.  Extrapolated to production
($M_{\max} \approx 12\,600$): ${\sim}4\times$, giving a total per-iteration cost
of ${\sim}280$\,s on 388 threads for a 2\,mm lens at $\alpha=30^\circ$.
